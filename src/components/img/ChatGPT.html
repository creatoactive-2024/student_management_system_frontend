To render the provided class schedule data in tabs within table columns using Ant Design in React, you can follow these steps:

  1. Extract the unique weekdays from the classSchedulesData and organize the classes accordingly.
  2. Create tabs for each weekday and display the class details in corresponding columns.
  3. Use Ant Design components like Tabs, Table, and Typography for structuring and displaying the data.

Here's how you can achieve this:

javascript
import React from 'react';
import { Tabs, Table, Typography } from 'antd';

const { TabPane } = Tabs;
const { Text } = Typography;

const ClassScheduleTable = ({ classSchedulesData }) => {
    // Extract weekdays from classSchedulesData
    const weekdays = Object.keys(classSchedulesData.schedules).reduce((acc, date) => {
        const weekday = new Date(date).toLocaleDateString('en-US', { weekday: 'long' });
        if (!acc.includes(weekday)) {
            acc.push(weekday);
        }
        return acc;
    }, []);

    // Render class schedule table columns
    const renderTableColumns = () => {
        const columns = [
            {
                title: 'Time',
                dataIndex: 'time',
                key: 'time',
            }
        ];

        // Add room columns dynamically
        Object.keys(classSchedulesData.schedules['2024-01-30']).forEach(room => {
            columns.push({
                title: room,
                children: [
                    {
                        title: 'Class',
                        dataIndex: room,
                        key: room,
                        render: (classes, record) => {
                            const classDetails = classes.find(cls => cls.class_time === record.time);
                            return classDetails ? (
                                <Text>
                                    {classDetails.classes.map(cls => (
                                        <div key={cls.class_level}>
                                            <b>Class Level:</b> {cls.class_level}<br />
                                            <b>Teacher:</b> {cls.teacher}<br />
                                            <b>Courses:</b> {cls.courses.join(', ')}<br />
                                            <b>Students:</b> {cls.students.join(', ')}
                                        </div>
                                    ))}
                                </Text>
                            ) : null;
                        }
                    }
                ]
            });
        });

        return columns;
    };

    // Render class schedule table data
    const renderTableData = () => {
        const data = [];
        // Assume classes have 15-minute intervals
        for (let hour = 0; hour < 24; hour++) {
            for (let minute = 0; minute < 60; minute += 15) {
                const time = `${hour.toString().padStart(2, '0')}:${minute.toString().padStart(2, '0')}`;
                const rowData = { time };

                // Fill data for each room
                Object.keys(classSchedulesData.schedules['2024-01-30']).forEach(room => {
                    rowData[room] = classSchedulesData.schedules['2024-01-30'][room].classes[weekday].filter(
                        cls => cls.class_time === time
                    );
                });

                data.push(rowData);
            }
        }
        return data;
    };

    return (
        <Tabs defaultActiveKey={weekdays[0]}>
            {weekdays.map(weekday => (
                <TabPane tab={weekday} key={weekday}>
                    <Table
                        columns={renderTableColumns()}
                        dataSource={renderTableData()}
                        bordered
                        pagination={false}
                        size="small"
                    />
                </TabPane>
            ))}
        </Tabs>
    );
};

export default ClassScheduleTable;

Ensure that you have imported the necessary Ant Design components and customize the rendering according to your UI requirements. Adjustments might be needed based on your specific layout and styling preferences.
